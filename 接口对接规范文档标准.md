function handleApiError(error) {
  switch (error.code) {
    case 4001:
      // 未登录，跳转到登录页
      wx.navigateTo({ url: '/pages/login/login' });
      break;
      
    case 4005:
      // 需要管理员权限
      wx.showModal({
        title: '权限不足',
        content: '此功能需要管理员权限',
        showCancel: false
      });
      break;
      
    case 1004:
      // 积分不足
      wx.showModal({
        title: '积分不足',
        content: `当前积分：${error.data.current_points}，需要：${error.data.required_points}`,
        confirmText: '去获取积分',
        success: (res) => {
          if (res.confirm) {
            wx.navigateTo({ url: '/pages/camera/camera' });
          }
        }
      });
      break;
      
    default:
      wx.showToast({
        title: error.msg || '操作失败',
        icon: 'none'
      });
  }
}
